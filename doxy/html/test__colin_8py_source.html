123 def connect(self):
124 """Establish socket connection with error handling"""
125 try:
126 if self.socket:
127 self.socket.close()
128 self.socket = socket.socket(socket.AF_INET, socket.SOCK_STREAM)
129 self.socket.connect((HOST, PORT))
130 self.theSbAccess = SBAccess(self.socket)
131 self.socket_open = True
132 self.is_connected = True
133 return True
134 except Exception as e:
135 print(f"Connection failed: {e}")
136 self.is_connected = False
137 self.socket_open = False
138 return False

313 def start_capture_monitoring(self):
314 try:
315 self.running = False
316 capture = self.theSbAccess.StartCapture('0000')
317
318 while True:
319 try:
320 if self.theSbAccess.IsCapturing() and self.theSbAccess.GetLastImageCaptured(capture) > 0:
321 break
322 except Exception:
323 if not self.ensure_connection():
324 return
325 time.sleep(0.02)
326
327 num_rows = self.theSbAccess.GetNumYRows(capture)
328 num_cols = self.theSbAccess.GetNumXColumns(capture)
329 num_timepoints = self.theSbAccess.GetNumTimepoints(capture)
330
331 prev_capture = -1
332 frame_count = 0
333
334 while True:
335 try:
336 if not self.theSbAccess.IsCapturing():
337 break
338
339 last_capture = self.theSbAccess.GetLastImageCaptured(capture)
340 if last_capture == prev_capture:
341 time.sleep(0.02)
342 continue
343
344 for tp in range(prev_capture + 1, last_capture + 1):
345 try:
346 image = self.theSbAccess.ReadImagePlaneBuf(capture, 0, tp, 0, 0)
347 print(f"Read timepoint: {tp}")
348 if len(image) == 0:
349 print("No more images")
350 break
351
352 if frame_count % 2 == 0:
353 self.ax.clear()
354 self.ax.imshow(image.reshape(num_rows, num_cols))
355 self.ax.set_title(f"Timepoint: {tp}")
356 self.canvas.draw()
357
358 frame_count += 1
359 except Exception:
360 if not self.ensure_connection():
361 return
362
363 prev_capture = last_capture
364 if last_capture >= num_timepoints - 1:
365 break
366
367 except Exception:
368 if not self.ensure_connection():
369 return
370 time.sleep(0.02)
371
372 finally:
373 self.capture_running = False
374 self.capture_button.config(text="Start Capture")
375 self.running = True
