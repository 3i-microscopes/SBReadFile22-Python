304 def RecvBigData(self,n):
305 # Helper function to recv n bytes or return None if EOF is hit
306 data = bytearray()
307 while len(data) < n:
308 packet = self.mSocket.recv(n - len(data))
309 if not packet:
310 return None
311 data.extend(packet)
312 return data

314 def Recv(self):
315
316 theRecvBuf = b''
317 b = self.mSocket.recv(1)
318 if b != b'&':
319 raise Exception("First character in answer must be a: &")
320 while True:
321 b = self.mSocket.recv(1)
322 if b == b'(':
323 continue
324 if b == b')':
325 break;
326
327 theRecvBuf += bytes(b)
328 # parse the string
329 str = bu.bytes_to_string(theRecvBuf)
330 #print("str is: ",str)
331 #split in list of arguments
332 largs = str.split(',')
333 #receive all the largs
334 if(len(largs) != 1):
335 raise Exception("Can only receive n values of same type")
336 arg = largs[0]
337
338 #split is size and format
339 prop = arg.split(":")
340 if(len(prop) != 2):
341 raise Exception("Invalid argument format: " + str)
342 theNum = int(prop[0])
343 theType = prop[1]
344
345 theSize = 1
346 if (theType == 'i4' or theType == 'u4' or theType == 'f4'):
347 theSize = 4
348 elif (theType == 'i2' or theType == 'u2'):
349 theSize = 2
350 elif (theType == 'i8' or theType == 'u8' or theType == 'f8'):
351 theSize = 8
352
353 theValBuf = b''
354 theValBuf = self.RecvBigData(theNum * theSize)
355 #print('theValBuf is: ',theValBuf)
356
357 if(len(theValBuf) != theNum * theSize):
358 raise Exception("Did not receive enough data")
359
360 if theType == 's':
361 theStr = bu.bytes_to_string(theValBuf)
362 return theStr
363 else:
364 theArr = bu.bytes_to_type(theValBuf,theType)
365 return theNum,theArr
