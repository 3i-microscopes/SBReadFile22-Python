88 def IsSFMT(self,inPath):
89 theStream = open(inPath,"rb")
90 theNpyHeader = CNpyHeader()
91 theRes = theNpyHeader.ParseNpyHeader(theStream)
92 self.mSingleTimepointFile = False
93 if not theRes:
94 return False,0
95 if len(theNpyHeader.mShape) == 3:
96 theNumTimepoints = theNpyHeader.mShape[0]
97 theRes = True
98 self.mSingleTimepointFile = True
99 if theNumTimepoints == 0:
100 theRes = False
101 self.mSingleTimepointFile = False
102 return theRes, theNumTimepoints
103 return False,0

106 def CountImageDataFiles(self):
107 theImageFileNames = self.mFile.GetListOfImageDataFiles(self.mImageTitle)
108 if self.mDebugPrint:
109 print ("theImageFileNames length: " , len(theImageFileNames))
110 print ("CountImageDataFiles: mImageRecord.mNumChannels " , self.mImageRecord.mNumChannels)
111 print ("CountImageDataFiles: mImageRecord.mNumTimepoints " , self.mImageRecord.mNumTimepoints)
112 if len(theImageFileNames) == self.mImageRecord.mNumChannels * self.mImageRecord.mNumTimepoints:
113 if not self.mImageRecord.mNumPlanes == 1:
114 # all in order
115 return True
116
117 #check for single file containing multi time points
118 if len(theImageFileNames) == self.mImageRecord.mNumChannels and self.mImageRecord.mNumPlanes == 1:
119 theNumTimepoints = 0
120 for theImageFile in theImageFileNames:
121 theShapeTP = 0
122 (theRes,theShapeTP) = self.IsSFMT(theImageFile)
123 if not theRes:
124 continue
125 if theShapeTP <= 0:
126 continue
127 # we are using min in case a channel has less timepoints than another one (crashed between channels)
128 if theNumTimepoints < theShapeTP:
129 theNumTimepoints = theShapeTP
130 if theNumTimepoints == 0:
131 theNumTimepoints = 1
132 self.mImageRecord.mNumTimepoints = theNumTimepoints
133 return True
134 # end of check for single file containing multi time points
135
136 #discrepancy
137
138 theMaxChannel = 0
139 theMaxTimepoint = 0
140 for theImageFile in theImageFileNames:
141 theChannel = self.mFile.GetChannelIndexOfPath(theImageFile)
142 theTimepoint = self.mFile.GetTimepointOfPath(theImageFile)
143
144 theMaxChannel = max(theMaxChannel, theChannel + 1)
145 theMaxTimepoint = max(theMaxTimepoint, theTimepoint + 1)
146
147 if self.mDebugPrint:
148 print ("CountImageDataFiles: theMaxChannel + theMaxTimepoint " , theMaxChannel , " " , theMaxTimepoint)
149 if theMaxChannel == 0 or theMaxTimepoint == 0:
150 print ("CountImageDataFiles: theMaxChannel + theMaxTimepoint " , theMaxChannel , theMaxTimepoint)
151 return False
152 self.mImageRecord.mNumTimepoints = theMaxTimepoint
153 self.mImageRecord.mNumChannels = theMaxChannel
154 self.mImageDataList = []
155 theTimepoint = 0
156 while theTimepoint < self.mImageRecord.mNumTimepoints:
157 #self.mImageDataList.append(CImageData())
158 theTimepoint += 1
159 if self.mDebugPrint:
160 #print ("CountImageDataFiles: mImageDataList.length " , len(self.mImageDataList))
161 print ("CountImageDataFiles: mImageDataList.length " , theTimepoint)
162 return True
